// =====================================================
// Google Apps Script — Google Sheets에 붙여넣기
// =====================================================
//
// [설정 방법]
// 1. Google Sheets 새로 만들기 (sheets.new)
// 2. 첫 행(헤더)에 아래 입력:
//    A1: 이름 | B1: 레벨 | C1: 등급 | D1: 레벨명
//    E1: Q1  | F1: Q2  | G1: Q3  | H1: Q4
//    I1: Q5  | J1: Q6  | K1: Q7  | L1: Q8
//    M1: 제출시간
//
// 3. 메뉴 → 확장 프로그램 → Apps Script
// 4. 기존 코드 지우고 아래 전체 붙여넣기
// 5. 배포 → 새 배포 → 유형: 웹 앱
//    - 실행 사용자: 나
//    - 액세스: 모든 사용자
// 6. 배포 URL 복사 → index.html의 GOOGLE_SCRIPT_URL에 붙여넣기
// =====================================================

function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data = JSON.parse(e.postData.contents);

  sheet.appendRow([
    data.name,
    data.level,
    data.grade,
    data.levelName,
    data.q1,
    data.q2,
    data.q3,
    data.q4,
    data.q5,
    data.q6,
    data.q7,
    data.q8,
    data.timestamp
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({ result: 'ok' }))
    .setMimeType(ContentService.MimeType.JSON);
}
